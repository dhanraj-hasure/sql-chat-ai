-- AI-Powered Portfolio Query: Find matching skills and projects
SELECT 
    u.full_name,
    u.email,
    u.role,
    s.skill_name,
    s.level AS skill_level,
    p.project_name,
    p.tech_stack,
    CONCAT(u.full_name, ' - ', p.project_name) AS portfolio_entry,
    CASE 
        WHEN s.level = 'Expert' THEN 'Highlight'
        ELSE 'Secondary'
    END AS emphasis
FROM users u
CROSS JOIN skills s
CROSS JOIN projects p
WHERE u.email = 'dhanrajdhasure@gmail.com'
    AND s.skill_name IN ('Java', 'Spring Boot', 'Angular')
    AND p.tech_stack LIKE '%Java%'
ORDER BY u.full_name, s.level DESC, p.project_name;
